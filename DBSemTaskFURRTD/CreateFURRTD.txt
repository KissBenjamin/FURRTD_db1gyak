CREATE TABLE Film(
filmId VARCHAR(3) PRIMARY KEY NOT NULL,
nev VARCHAR (20) NOT NULL,
terem int VARCHAR(2)NOT NULL,
rendezo VARCHAR(20) NOT NULL,,
hossz int,
műfaj VARCHAR(10)NOT NULL,);

CREATE TABLE Néző(
gyerek int,
diák int,
felnőtt int);

CREATE TABLE Mozi(
moziID VARCHAR(3) PRIMARY KEY NOT NULL,
település VARCHAR(30) NOT NULL);

CREATE TABLE Beaosztás(
esti,
délutáni,
délelőtti);

CREATE TABLE Takarítók(
takaritoID VARCHAR(3) PRIMARY KEY NOT NULL,
nev VARCHAR(20) NOT NULL);
